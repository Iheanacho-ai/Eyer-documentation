# Troubleshooting and Support

import React, { useEffect } from 'react';
import { faq } from './js/questions.js';

export const Section = () => {
  useEffect(() => {
    const acc = document.getElementsByClassName("accordion");

    const handleAccordionClick = (event) => {
      event.currentTarget.classList.toggle("active");
      const panel = event.currentTarget.nextElementSibling;
      const arrow = event.currentTarget.querySelector(".accordion-span");

      if (panel.style.display === "block") {
        panel.style.display = "none";
        arrow.classList.remove("active");
      } else {
        panel.style.display = "block";
        arrow.classList.add("active");
      }
    };

    for (let i = 0; i < acc.length; i++) {
      acc[i].addEventListener("click", handleAccordionClick);
    }

    // Cleanup event listeners on component unmount
    return () => {
      for (let i = 0; i < acc.length; i++) {
        acc[i].removeEventListener("click", handleAccordionClick);
      }
    };
  }, []);



  return (
    <div>
      {faq.map((item, index) => (
        <div className="accordion-container" key={index}>
          <div className="accordion">{item.q} <span className="accordion-span">&#x3e;</span></div>
          <div className="panel">
            <p>{item.a}</p>
          </div>
        </div>
      ))}
    </div>
  );
};



If you are having any issues, please do not hesitate to reach out to us via our [support email](https://customer.support.eyer.ai/servicedesk/customer/portal/1/create/1) or our [Discord](https://discord.gg/c2nuGEGPs2) channel.
